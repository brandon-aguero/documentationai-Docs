## Implementation Example

<Tabs>
  <Tab title="Web Dashboard" icon="monitor">
    ````javascript
// Initialize patient management dashboard
const dashboard = new Neuroglia.PatientDashboard({
  container: '#patient-view',
  patientId: 'P12345'
});

// Load patient data
dashboard.loadPatient().then(() => {
  dashboard.displayVitalSigns();
  dashboard.showCarePlan();
  dashboard.enableNotifications();
});
    ````
  </Tab>
  <Tab title="Mobile App" icon="smartphone">
    ````javascript
import { PatientManager } from 'neuroglia-mobile';

const manager = new PatientManager({
  patientId: 'P12345',
  offlineMode: true
});

// Sync patient data
await manager.syncData();

// Update care plan
await manager.updateCarePlan({
  medications: ['Aspirin 81mg daily'],
  followUp: '2024-02-15'
});
    ````
