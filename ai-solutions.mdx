## Code Example

<Tabs>
  <Tab title="Node.js" icon="code">
    ````javascript
const neuroglia = require('neuroglia-ai');

const diagnostic = new neuroglia.DiagnosticAI({
  model: 'radiology-v2',
  confidenceThreshold: 0.85
});

// Analyze medical image
const result = await diagnostic.analyzeImage({
  imageUrl: 'https://hospital.example.com/images/xray001.jpg',
  patientId: 'P12345',
  modality: 'chest-xray'
});

console.log(result.findings);
// Output: [{ condition: 'pneumonia', confidence: 0.92, location: 'right-lung' }]
````
  </Tab>
  <Tab title="Python" icon="code">
    ````python
from neuroglia_ai import DiagnosticAI

diagnostic = DiagnosticAI(
    model='radiology-v2',
    confidence_threshold=0.85
)

result = diagnostic.analyze_image(
    image_url='https://hospital.example.com/images/xray001.jpg',
    patient_id='P12345',
    modality='chest-xray'
)

print(result.findings)
# Output: [{'condition': 'pneumonia', 'confidence': 0.92, 'location': 'right-lung'}]
````
